# First, we will inherit from pytorch Module class; https://pytorch.org/docs/stable/generated/torch.nn.Module.html

class Encoder(nn.Module):
    def __init__(self, seed=42): # Changed to take inputs as of class
        """
        seed: int, random seed for reproducibility
        """
        super(Encoder, self).__init__()
        self.seed = torch.manual_seed(seed)
        
        # here we are initialising the linear layers. This registeres the layer parameters (W,b) as parameters of the Module
        self.fc1 = nn.Linear(INPUT_SIZE,HIDDEN_SIZE)
        self.fc2 = nn.Linear(HIDDEN_SIZE,LATENT_SIZE)

        # We will initialise weights following a multivariate Gaussian
        nn.init.normal_(self.fc1.weight, mean=0, std=0.01)
        nn.init.normal_(self.fc1.bias, mean=0, std=0.01)
        nn.init.normal_(self.fc2.weight, mean=0, std=0.01)
        nn.init.normal_(self.fc2.bias, mean=0, std=0.01)

    # this defines a forward pass of the network (="applying" the network to some input data)
    def forward(self, x):
        x      = torch.nn.LeakyReLU()(self.fc1(x)) # Hidden with activation
        z      = self.fc2(x)
        return z

class Decoder(nn.Module):
    def __init__(self, seed=42): # Changed from no inputs from class
        """
        seed: int, random seed for reproducibility
        """
        super(Decoder, self).__init__()
        self.seed = torch.manual_seed(seed)
        # Using the encoder, how will you initialize two layers?
        self.fc1 = nn.Linear(LATENT_SIZE,HIDDEN_SIZE)
        self.fc2 = nn.Linear(HIDDEN_SIZE,INPUT_SIZE)    

        # Using the endoer, how will you initialize the weights and biases?
        # Initialise weights and biases using same as encoder?
        nn.init.normal_(self.fc1.weight, mean=0, std=0.01)
        nn.init.normal_(self.fc1.bias, mean=0, std=0.01)
        nn.init.normal_(self.fc2.weight, mean=0, std=0.01)
        nn.init.normal_(self.fc2.bias, mean=0, std=0.01)
        

    def forward(self, z):
        # Using the encoder, how will you define the forward pass?
        # I recommend the same activation function!
        x = torch.nn.LeakyReLU()(self.fc1(z))  # Apply LeakyReLU to the first layer's output
        x = self.fc2(x)          # Reconstruct the input
        
        return x

class Autoencoder(nn.Module):
    def __init__(self):
        super(Autoencoder, self).__init__()
        # here we are creating instances of the Encoder and Decoder class
        self.encoder = Encoder()
        self.decoder = Decoder()

    def forward(self, x):
        z = self.encoder(x)
        x = self.decoder(z)
        return x

Encoder take sinput data and compresses to hidden size and then to latent size.
For our decoder we want to take the output (latent size) from the encoder and reconstruct original input data.
Initialise two layers with opposite layers to encoder so we can reconstruct.
Initialise weights and biases the same way as encoder (unsure)
Define forward as reverse of encoder. Pass z through first layer and apply Leaky ReLU. Pass output of first layer through second layer.

# the training loop takes a function that loads the data batch by batch, a model to train, a loss function to train the model on and an optimizer
def train_loop(dataloader, model, loss_fn, optimizer):
    size = len(dataloader.dataset)
    losses = []
    # iterate over the dataset
    for batch, (X, y) in enumerate(dataloader):
        # Compute prediction of the model
        X = X.float() # Convert to avoid error
        pred = model(X)
        # Compute the loss function
        loss = loss_fn(pred,X)

        # Backpropagation; this is where we take the gradient and update the network parameters
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        # here we keep track of the loss
        if batch % 100 == 0: # Allow for batches not 1st to print and track
            loss, current = loss.item(), batch * len(X)
            losses.append(loss)
            #print(f"loss: {loss:>7f}  [{current:>5d}/{size:>5d}]")

    return losses

# the test loop is similar to the training loop, only that we don't take any gradients/don't update the network parameters, but only evaluate
def test_loop(dataloader, model, loss_fn):
    size = len(dataloader.dataset)
    num_batches = len(dataloader)
    test_loss =  0

    model.eval() # From class
    with torch.no_grad():
        for (X, y) in dataloader:
            X = X.float() # Fix error
            pred = model(X)
            #test_loss += loss_fn(pred, y).item()
            test_loss += loss_fn(pred, X).item() # Switched from class

    test_loss /= num_batches
    #print(f" Avg loss: {test_loss:>8f} \n")

    return test_loss

def make_ellipses(gmm, ax, my_var1 = 0, my_var2 = 1):
    """
    Primarily stolen from: https://scikit-learn.org/stable/auto_examples/mixture/plot_gmm_covariances.html
    gmm: GaussianMixture object from sklearn
    ax: Matplotlib axis, indication of where to plot the contour
    my_var1: int, x-axis in the latent space where you are plotting
    my_var2: int, y-axis in the latent space where you are plotting
    """
    for n in np.arange(len(gmm.covariances_)):
        covariances = gmm.covariances_[n][[[my_var1,my_var1],[my_var1, my_var2]],[[my_var1,my_var2], [my_var2,my_var2]]]
        v, w = np.linalg.eigh(covariances) #Returns the eigenvalues / vectors of our covariance matrix
        u = w[0] / np.linalg.norm(w[0])
        angle = np.arctan2(u[1], u[0])
        angle = 180 * angle / np.pi  # convert to degrees
        v = 2.0 * np.sqrt(2.0) * np.sqrt(v)
        ell = mpl.patches.Ellipse(
            gmm.means_[n, [my_var1, my_var2]], v[0], v[1], angle=180 + angle, edgecolor='k', lw=1, facecolor='none'
        )
        ell.set_clip_box(ax.bbox)
        ell.set_alpha(0.5)
        ax.add_artist(ell)
