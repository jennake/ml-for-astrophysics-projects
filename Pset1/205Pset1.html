def my_model(theta, x):
    # Takes theta: vector of weights
    # Takes x: vector of features
    # Compute dot product  and x
    dotprod = np.dot(x, theta)
    # Apply sigmoid
    probability_Ia = 1 / (1 + np.exp(-dotprod))
    
    return probability_Ia

#1d Objective Function/Cross Entropy Score:

def my_objective_function(theta, x, y_true, smalle = 10 **(-10)):
    
    # Use model to predict y and then compare with true y
    y_predicted = my_model(theta, x)
    
    # Use clip to prevent 0 errors 
    y_spredicted = np.clip(y_predicted, smalle, 1 - smalle)
    
    # Use definition of CE
   
    CE = -np.sum((y_true * np.log(y_spredicted)+(1-y_true)*np.log(1-y_spredicted)))
    cross_entropy_score = CE/len(y_true)
    return cross_entropy_score

# Objective Function: Negative Log-Likelihood 
# Simplified as we did in class
def neg_log_likelihood(H_0, v, distance, distance_uncertainty):
    h_distance = hub_model(v, H_0)
    chi_squared = ((distance - h_distance) ** 2) / (2 * distance_uncertainty ** 2)
    log_likelihood = np.sum(chi_squared + 0.5 * np.log(2 * np.pi * distance_uncertainty ** 2))
    return log_likelihood

# Define derivative of nll to then optimise
def nll_derivative(H_0, v, distance, distance_uncertainty, epsilon=1e-5):
    return (neg_log_likelihood(H_0 + epsilon, v, distance, distance_uncertainty) - 
            neg_log_likelihood(H_0 - epsilon, v, distance, distance_uncertainty)) / (2 * epsilon)

# Use Gradient Descent without momentum 
def gradient_descent(v, distance, distance_uncertainty, H0_init=70, learning_rate=0.01, epochs=1000, tol=1e-5, max_iter=1000):
    H_0 = H0_init  # Initial guess
    for i in range(max_iter): # Iterate
        grad = nll_derivative(H_0, v, distance, distance_uncertainty) # Gradient of NLL
        H_0 = H_0 - (learning_rate * grad) # Update H_0, opposite to gradient to find min
        if abs(grad) < tol:  # Stop if gradient is small as it does not change much (5dp)
            break
    return H_0

def mgradient_descent(v, distance, distance_uncertainty, H0_init=70, learning_rate=0.01, decay_rate = 0.5, epochs=1000, tol=1e-5, max_iter=1000):
    H_0 = H0_init  # Initial guess
    M_0 = 0 # Initial momentum
    for i in range(max_iter): # Iterate
        grad = nll_derivative(H_0, v, distance, distance_uncertainty) # Gradient of NLL
        M_0 = M_0 * decay_rate # Now introduce momentum, using decay rat e as another parameter
        H_0 = H_0 - (learning_rate * grad) + M_0 # Update H_0 now with momentum included
        if abs(grad) < tol:  # Stop if gradient is small as it does not change much (5dp)

    return H_0

# Gradient Descent with Momentum
def mgradient_descent(v, distance, distance_uncertainty, H0_init=70, learning_rate=0.01, decay_rate = 0.8, epochs=1000, tol=1e-5):
    H_0 = H0_init  # Initial guess
    M_0 = 0 # Initial momentum
    histm  = []
    for i in range(epochs): # Iterate
        grad = nll_derivative(H_0, v, distance, distance_uncertainty) # Gradient of NLL
        #M_0 = M_0 * decay_rate # Now introduce momentum, using decay rate as another parameter
        M_0 = decay_rate * M_0 + learning_rate * grad
        H_0 = H_0 - M_0
        histm.append(neg_log_likelihood(H_0, v, distance, distance_uncertainty))
        if abs(grad) < tol:  # Stop if gradient is small as it does not change much (5dp)
            break
    return H_0, histm
